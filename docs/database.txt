CREATE FUNCTION Distance(latitud1 DOUBLE, longitud1 DOUBLE, latitud2 DOUBLE, longitud2 DOUBLE) RETURNS DOUBLE
    DETERMINISTIC
BEGIN
    DECLARE r DOUBLE DEFAULT 6371000;
    DECLARE phi1 DOUBLE DEFAULT latitud1 * PI() / 180;
    DECLARE phi2 DOUBLE DEFAULT latitud2 * PI() / 180;
    DECLARE deltaPhi DOUBLE DEFAULT (latitud2 - latitud1) * PI() / 180;
    DECLARE deltaLambda DOUBLE DEFAULT (longitud2 - longitud1) * PI() / 180;
    DECLARE a DOUBLE;
    DECLARE c DOUBLE;
    SET a = SIN(deltaPhi / 2) * SIN(deltaPhi / 2) + COS(phi1) * COS(phi2) * SIN(deltaLambda / 2) * SIN(deltaLambda / 2);
    SET c = 2 * ATAN2(SQRT(a), SQRT(1 - a));
    RETURN r * c;
END;

DROP TABLE IF EXISTS UserBusiness;
DROP TABLE IF EXISTS BusinessAddress;
DROP TABLE IF EXISTS BusinessCategory;
DROP TABLE IF EXISTS BusinessHours;
DROP TABLE IF EXISTS BusinessImage;
DROP TABLE IF EXISTS BusinessPhoneNumber;
DROP TABLE IF EXISTS Category;
DROP TABLE IF EXISTS Country;
DROP TABLE IF EXISTS City;
DROP TABLE IF EXISTS State;
DROP TABLE IF EXISTS ReportReview;
DROP TABLE IF EXISTS Report;
DROP TABLE IF EXISTS Session;
DROP TABLE IF EXISTS User;
DROP TABLE IF EXISTS Business;

CREATE TABLE IF NOT EXISTS User
(
    id           VARCHAR(36),
    fullname     VARCHAR(70)                         NOT NULL,
    email        VARCHAR(255)                         NOT NULL,
    phoneNumber  VARCHAR(15),
    thumbnailUrl VARCHAR(500),
    password     VARCHAR(256),
    typeLogin    ENUM ('email', 'facebook', 'google') NOT NULL,
    typeUser     ENUM ('moderator', 'normal')         NOT NULL,
    createdAt    DATETIME                             NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updatedAt    DATETIME                             NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    deletedAt    DATETIME                                      DEFAULT NULL,
    PRIMARY KEY (id),
    UNIQUE INDEX (email)
);

create table IF NOT EXISTS Session
(
    token     varchar(36) not null,
    userId    varchar(36) not null,
    createdAt DATETIME    NOT NULL DEFAULT CURRENT_TIMESTAMP,
    deletedAt DATETIME             DEFAULT NULL,
    primary key (token),
    foreign key (userId) references User (id) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS Business
(
    id          VARCHAR(36),
    name        VARCHAR(80),
    bannerUrl   VARCHAR(500),
    description VARCHAR(500),
    createdAt   DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updatedAt   DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    deletedAt   DATETIME          DEFAULT NULL,
    PRIMARY KEY (id)
);

CREATE TABLE IF NOT EXISTS UserBusiness
(
    userId     VARCHAR(36) NOT NULL,
    businessId VARCHAR(36) NOT NULL,
    FOREIGN KEY (userId) REFERENCES User (id) ON UPDATE CASCADE ON DELETE CASCADE,
    FOREIGN KEY (businessId) REFERENCES Business (id) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS UserReview
(
    userId      VARCHAR(36)      NOT NULL,
    businessId  VARCHAR(36)      NOT NULL,
    rating      TINYINT UNSIGNED NOT NULL,
    description VARCHAR(500),
    FOREIGN KEY (userId) REFERENCES User (id) ON UPDATE CASCADE ON DELETE CASCADE,
    FOREIGN KEY (businessId) REFERENCES Business (id) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS BusinessHours
(
    businessId VARCHAR(36)       NOT NULL,
    day        TINYINT           NOT NULL, # 1 Monday 2 Tuesday ... 7 Sunday
    openTime   SMALLINT UNSIGNED NOT NULL,
    closeTime  SMALLINT UNSIGNED NOT NULL,
    FOREIGN KEY (businessId) REFERENCES Business (id) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS BusinessImage
(
    businessId VARCHAR(36)  NOT NULL,
    imageUrl   VARCHAR(500) NOT NULL,
    FOREIGN KEY (businessId) REFERENCES Business (id) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS BusinessPhoneNumber
(
    businessId  VARCHAR(36) NOT NULL,
    phoneNumber VARCHAR(50),
    FOREIGN KEY (businessId) REFERENCES Business (id) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS Category
(
    code     INT UNSIGNED NOT NULL AUTO_INCREMENT,
    category VARCHAR(250),
    PRIMARY KEY (code)
);

CREATE TABLE IF NOT EXISTS BusinessCategory
(
    businessId   VARCHAR(36)  NOT NULL,
    categoryCode INT UNSIGNED NOT NULL,
    FOREIGN KEY (businessId) REFERENCES Business (id) ON UPDATE CASCADE ON DELETE CASCADE,
    FOREIGN KEY (categoryCode) REFERENCES Category (code) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS Report
(
    id          VARCHAR(36),
    userId      VARCHAR(36),
    businessId  VARCHAR(36),
    title       VARCHAR(250),
    description VARCHAR(500),
    createdAt   DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updatedAt   DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    deletedAt   DATETIME          DEFAULT NULL,
    reviewedAt  DATETIME          DEFAULT NULL,
    PRIMARY KEY (id),
    FOREIGN KEY (userId) REFERENCES User (id) ON UPDATE CASCADE ON DELETE CASCADE,
    FOREIGN KEY (businessId) REFERENCES Business (id) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS ReportReview
(
    userId   VARCHAR(36) NOT NULL,
    reportId VARCHAR(36) NOT NULL,
    analysis VARCHAR(200),
    createdAt DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (userId) REFERENCES User (id) ON UPDATE CASCADE ON DELETE CASCADE,
    FOREIGN KEY (reportId) REFERENCES Report (id) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS Country
(
    code INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(80),
    PRIMARY KEY (code)
);

CREATE TABLE IF NOT EXISTS State
(
    code int         NOT NULL AUTO_INCREMENT,
    name varchar(80) NOT NULL,
    countryCode INT NOT NULL,
    PRIMARY KEY (code),
    FOREIGN KEY (countryCode) REFERENCES Country(code) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS City
(
    code      INT         NOT NULL AUTO_INCREMENT,
    stateCode INT         NOT NULL,
    name      varchar(80) NOT NULL,
    PRIMARY KEY (code),
    FOREIGN KEY (stateCode) REFERENCES State (code) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS BusinessAddress
(
    id          VARCHAR(36),
    businessId  VARCHAR(36) NOT NULL,
    address     VARCHAR(250),
    cityCode    INT,
    stateCode   INT,
    countryCode INT,
    latitude    DOUBLE,
    longitude   DOUBLE,
    PRIMARY KEY (id),
    FOREIGN KEY (businessId) REFERENCES Business (id) ON UPDATE CASCADE ON DELETE CASCADE,
    FOREIGN KEY (countryCode) REFERENCES Country (code) ON UPDATE CASCADE ON DELETE CASCADE,
    FOREIGN KEY (stateCode) REFERENCES State (code) ON UPDATE CASCADE ON DELETE CASCADE,
    FOREIGN KEY (cityCode) REFERENCES City (code) ON UPDATE CASCADE ON DELETE CASCADE
);
